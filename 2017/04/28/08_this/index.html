this

this 指调用函数的那个对象

纯粹函数调用

通常的函数调用属于全局性调用，this指代全局对象Global。

var x = 1;
function test() {
    this.x = 0;
}
test(); 
console.log(this.x); //0

作为对象方法调用

函数还可以对象的方法调用，this指代这个上级对象

function test(){
    console.log(this.x);
}
var o = {};
o.x = 1;
o.m = test;
o.m(); // 1

作为构造函数调用

this指代构造函数生成的新对象

function test() {
    this.x = 1;
}
var o = new test();
console.log(o.x);//1

改变函数的调用对象的方法调用

apply

apply()是函数对象的一个方法，它的作用是改变函数的调用对象，第一个参数表示改变后调用这个函数的对象，this指代第一个参数，参数为空时默认为全局对象。

var x = 0;
function test() {
    console.log(this.x);
}
var o = {};
o.x = 1;
o.m = test();
o.m.apply(); //0
o.m.apply(o); //1

call

call和apply的作用一样，只是接受参数的方式不一样

var func = function(arg1, arg2) {};
// 调用
func.apply(this, [arg1, arg2]);
func.call(this, arg1, arg2);

当传入参数个数不确定时最好用apply，明确固定个数参数时一般用call，call还用来验证类型，转换伪数组为标准数组

function log() {
    console.log.apply(console, arguments);
} 
log(1); //1
log(1, 2); //1 2

function getObjType(obj) {
    return Object.prototype.toString().call(obj);
}

function log1() {
    var args = Array.prototype.slice.call(arguments);
    args.unshift('(app)');
    console.log.apply(console, args);
}

bind

MDN的解释是：bind()方法会创建一个新函数，称为绑定函数，当调用这个绑定函数时，绑定函数会以创建它时传入 bind()方法的第一个参数作为 this，传入 bind() 方法的第二个以及以后的参数加上绑定函数运行时本身的参数按照顺序作为原函数的参数来调用原函数。

var bar = function(){
    console.log(this.x);
}
var foo = {
    x:3
}
bar(); // undefined
var func = bar.bind(foo);
func(); // 3

在Javascript中，多次 bind() 是无效的。更深层次的原因， bind() 的实现，相当于使用函数在内部包了一个 call / apply ，第二次 bind() 相当于再包住第一次 bind() ,故第二次以后的 bind 是无法生效的。

var bar = function(){
    console.log(this.x);
};
var foo = {
    x:3
};
var sed = {
    x:4
};
var func = bar.bind(foo).bind(sed);
func(); //3

var fiv = {
    x:5
};
var func = bar.bind(foo).bind(sed).bind(fiv);
func(); //3

apply\call\bind区别

当希望改变上下文环境之后并非立即执行，而是回调执行的时候，使用 bind() 方法。而 apply/call 则会立即执行函数。
